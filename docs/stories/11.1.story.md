## Story

**As a** visitante ou aluno autenticado,
**I want** clicar em "Quero me matricular" ou "Assinar plano" e ser redirecionado para o checkout Stripe,
**so that** eu consiga comprar um curso avulso ou assinar um plano de mentoria de forma segura e sem sair da plataforma.

## Business Value

O gateway adapter (Story 10.1) criou a interface `CheckoutProvider` e o `stubProvider` — o plug-in point está pronto. Esta story ativa o provider Stripe real: cria o `stripeProvider`, expõe as API routes `/api/checkout/curso` e `/api/checkout/plano`, e substitui o redirect de login no `CheckoutButton` pela chamada real à API. Sem isso, nenhum aluno consegue pagar — a plataforma existe mas não gera receita.

### stripeProvider — implementação

```typescript
// src/lib/checkout/providers/stripe.ts
import Stripe from 'stripe'
import type {
  CheckoutProvider,
  CreateCourseCheckoutParams,
  CreatePlanCheckoutParams,
  CheckoutSession,
} from '../types'

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
  apiVersion: '2024-06-20',
})

export const stripeProvider: CheckoutProvider = {
  async createCourseCheckout(params: CreateCourseCheckoutParams): Promise<CheckoutSession> {
    const session = await stripe.checkout.sessions.create({
      mode: 'payment',
      customer_email: params.userEmail,
      line_items: [
        {
          quantity: 1,
          price_data: {
            currency: 'brl',
            unit_amount: params.priceAmountCents,
            product_data: { name: params.courseTitle },
          },
        },
      ],
      metadata: {
        type: 'course',
        courseId: params.courseId,
        userId: params.userId,
      },
      success_url: params.successUrl,
      cancel_url: params.cancelUrl,
    })
    return { url: session.url! }
  },

  async createPlanCheckout(params: CreatePlanCheckoutParams): Promise<CheckoutSession> {
    // Planos usam Price IDs pré-configurados no Stripe Dashboard
    const priceId =
      params.billingPeriod === 'annual'
        ? process.env[`STRIPE_PRICE_${params.planId.toUpperCase()}_ANNUAL`]!
        : process.env[`STRIPE_PRICE_${params.planId.toUpperCase()}_MONTHLY`]!

    const session = await stripe.checkout.sessions.create({
      mode: 'subscription',
      customer_email: params.userEmail,
      line_items: [{ price: priceId, quantity: 1 }],
      metadata: {
        type: 'plan',
        planId: params.planId,
        userId: params.userId,
        billingPeriod: params.billingPeriod,
      },
      success_url: params.successUrl,
      cancel_url: params.cancelUrl,
    })
    return { url: session.url! }
  },
}
```

### API route `/api/checkout/curso`

```typescript
// src/app/api/checkout/curso/route.ts
import { NextResponse } from 'next/server'
import { createClient } from '@/lib/supabase/server'
import { getCheckoutProvider } from '@/lib/checkout'
import { z } from 'zod'

const bodySchema = z.object({
  courseId: z.string().uuid(),
  billingPeriod: z.enum(['monthly', 'annual']).optional(),
})

export async function POST(request: Request) {
  const supabase = createClient()
  const {
    data: { user },
  } = await supabase.auth.getUser()
  if (!user) return NextResponse.json({ error: 'Não autenticado' }, { status: 401 })

  const parsed = bodySchema.safeParse(await request.json())
  if (!parsed.success) return NextResponse.json({ error: 'Dados inválidos' }, { status: 400 })

  const { courseId } = parsed.data

  const { data: course } = await supabase
    .from('courses')
    .select('id, title, price, slug')
    .eq('id', courseId)
    .eq('published', true)
    .single()

  if (!course) return NextResponse.json({ error: 'Curso não encontrado' }, { status: 404 })

  const courseRow = course as { id: string; title: string; price: number; slug: string }
  const baseUrl = process.env.NEXT_PUBLIC_APP_URL ?? 'http://localhost:3000'

  const { url } = await getCheckoutProvider().createCourseCheckout({
    courseId: courseRow.id,
    courseTitle: courseRow.title,
    userId: user.id,
    userEmail: user.email!,
    priceAmountCents: Math.round((courseRow.price ?? 0) * 100),
    successUrl: `${baseUrl}/checkout/sucesso?course=${courseRow.slug}`,
    cancelUrl: `${baseUrl}/cursos/${courseRow.slug}`,
  })

  return NextResponse.json({ url })
}
```

### CheckoutButton — atualizado

```typescript
// src/components/checkout/CheckoutButton.tsx
'use client'

import { useState } from 'react'
import { createClient } from '@/lib/supabase/client'

interface CheckoutButtonProps {
  label: string
  courseId?: string      // UUID do curso (para /api/checkout/curso)
  courseSlug?: string    // slug (para redirect de login)
  planId?: string        // id do plano (para /api/checkout/plano)
  billingPeriod?: 'monthly' | 'annual'
  className?: string
}

export function CheckoutButton({ label, courseId, courseSlug, planId, billingPeriod = 'monthly', className }: CheckoutButtonProps) {
  const [loading, setLoading] = useState(false)

  async function handleClick() {
    setLoading(true)
    const supabase = createClient()
    const { data: { session } } = await supabase.auth.getSession()

    // Não autenticado → redirect para login
    if (!session) {
      const redirect = courseSlug ? `/cursos/${courseSlug}` : planId ? `/planos` : '/'
      window.location.href = `/auth/login?redirect=${encodeURIComponent(redirect)}`
      return
    }

    // Autenticado → chamar API de checkout
    try {
      const endpoint = courseId ? '/api/checkout/curso' : '/api/checkout/plano'
      const body = courseId
        ? { courseId }
        : { planId, billingPeriod }

      const res = await fetch(endpoint, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(body),
      })
      const data = await res.json() as { url?: string; error?: string }
      if (data.url) {
        window.location.href = data.url
      } else {
        alert(data.error ?? 'Erro ao iniciar checkout')
      }
    } catch {
      alert('Erro de conexão. Tente novamente.')
    } finally {
      setLoading(false)
    }
  }

  return (
    <button
      onClick={() => { void handleClick() }}
      disabled={loading}
      className={className}
    >
      {loading ? 'Aguarde...' : label}
    </button>
  )
}
```
