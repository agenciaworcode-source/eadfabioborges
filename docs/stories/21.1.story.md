```typescript
const buildProgress = (enrollment: Enrollment, progressRecords: LessonProgress[]) => {
  const courseProgress = progressRecords.filter((p) => p.course_id === enrollment.course_id)
  const completedLessonIds = new Set(courseProgress.map((p) => p.lesson_id))

  const allLessons = enrollment.courses.modules.flatMap((m) => m.lessons)
  const totalLessons = allLessons.length
  const completedLessons = allLessons.filter((l) => completedLessonIds.has(l.id)).length
  const progressPercent = totalLessons > 0 ? Math.round((completedLessons / totalLessons) * 100) : 0
  const lastActivityAt = courseProgress[0]?.completed_at ?? null // jĂ¡ ordenado DESC

  const modules = enrollment.courses.modules.map((module) => {
    const moduleLessons = module.lessons
    const moduleCompleted = moduleLessons.filter((l) => completedLessonIds.has(l.id)).length
    return {
      module_id: module.id,
      module_title: module.title,
      module_order: module.order,
      total_lessons: moduleLessons.length,
      completed_lessons: moduleCompleted,
      progress_percent:
        moduleLessons.length > 0 ? Math.round((moduleCompleted / moduleLessons.length) * 100) : 0,
      lessons: moduleLessons.map((l) => ({
        lesson_id: l.id,
        lesson_title: l.title,
        is_completed: completedLessonIds.has(l.id),
      })),
    }
  })

  return { ...enrollment, totalLessons, completedLessons, progressPercent, lastActivityAt, modules }
}
```
